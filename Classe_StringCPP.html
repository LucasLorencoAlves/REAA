Classe string



Em C, a manipulação de uma string acontece por meio de uma sequência de caracteres disposta um caractere em cada memória como é o caso abaixo. 

                        
                            int mai()
                            {

                                char nome[6]="Lucas";
                                printf("%c", nome[0]);//L está em algum endereço de memória 
                                printf("%c", nome[1]);//u está em algum endereço de memória 
                                printf("%c", nome[2]);//c está em algum endereço de memória
                                printf("%c", nome[3]);//a está em algum endereço de memória 
                                printf("%c", nome[4]);//s está em algum endereço de memória 


                            }

Este estilo de string em C, é preciso ter cuidado, pois como a manipulação é por responsabilidade do programador, então se fizer algum acesso fora da matriz. Vai acabar invadindo outros espaço da memória ocasionando algum distúrbio no funcionamento do programa.

Mas! Em C++ a classe string se encarrega de tratar dessas manipulações sozinha, como é o caso da inclusão do valor nulo no final da stringe entre outras. Para cada string declarada ela se torna um objeto da classe string podendo assim chamar funções membros da classe. Mas também podemos tratara string como um vetor de dados da mesma forma que tratamos um array de caractere em C. Veja 

        
                            #include <string>
                            using std::string;

                            int main()
                            {

                                string str_1("Lucas");
                                cout << str_1 << endl;

                                cout << str_1[0] << endl;//Pode se manipulada caractere por caractere
                                cout << str_1[1] << endl;//idem
                                cout << str_1[2] << endl;//idem
                                cout << str_1[3] << endl;//idem
                                cout << str_1[4] << endl;//idem
                                
                            }

Se tratar a string como um objeto, podemos chamar funções membros para nos retornar um tamanho, comprimento e até mesmo nos retornar um valor de comparação entre duas strings. Como é mostrado no código abaixo. 

    
        #include <iostream>
        using std::cout;
        using std::endl;
        
        #include <string>
        using std::string;
        
        int main()
        {
            cout << "Hello REAA!" << endl;
        
            //Formas declaração de uma string
            string str_1="Meu nome eh ";
            string str_2("Lucas Lorenco Alves");
            string str_3=" ";
            cout << str_1 << str_2 << endl;
        
            //Posso concatenar  usando o operador sobrecarregado
            str_3= str_1 + str_2;
            cout << str_3 << endl;
        
            str_3+="!!!";
            cout << str_3 << endl;
        
            str_3.swap(str_1);
            cout << str_3 << endl;
        
            //======================================
            //Comparaçao de strings
            string  Alpha ("Alpha"),
                    Ohmega("Ohmega"),
                    Delta ("Ohmega");
        
            cout << "1° Retorno: " << (Alpha == Ohmega)     << endl;// Retorna 0 se forem diferente, senão 1
            cout << "2° Retorno: " << (Ohmega == Delta)     << endl;
            cout << "3° Retorno: " << (Alpha != Ohmega)     << endl;
            cout << "4° Retorno: " << (Alpha > Ohmega)      << endl;
            cout << "5° Retorno: " << (Alpha < Ohmega)      << endl;
            cout << "6° Retorno: " << (Alpha <= Ohmega)     << endl;
            cout << "7° Retorno: " << Alpha.compare(str_2)  << endl;
            cout << "8° Retorno: " << Ohmega.compare(Delta) << endl;
        
        
            cout << "9° Tamanho: " << Alpha.size() << endl;
            cout << "10° capacity: " << Ohmega.capacity() << endl;
            cout <<"11° comprimento: " << Ohmega.length() <<endl;
            cout <<"12° Size: " << Ohmega.size() <<endl;
        
            int tamanho= Ohmega.size();//Retorna o tamamnho da string Ohmega e atribui o valor na variavel tamanho;
            cout <<"Tamanho da string: "<< tamanho << endl;
        
            Alpha = Ohmega;//Atribui a string em Ohmega para Alpha
            cout << Alpha << endl;//atribui a string ohmega na string alpha 

            string  a = ("Alice e Beto gostam de "),
                    b = ("chocolate."),
                    c = ("doce de leite."),
                    d = ("pipoca."),
                    e = (c);

                cout << a + b << endl;
                cout << a + c << endl;
                cout << a + d << endl;
                cout << a + e << endl;
            return 0;
        }
