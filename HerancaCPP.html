Herança.



A herança é um conceito fundamental na programação orientada a objetos (POO) que permite a criação de hierarquias de classes, facilitando a reutilização de código e a modelagem de relacionamentos entre objetos. Em C++, a herança é implementada por meio da palavra-chave class e do modificador public ou private, determinando a visibilidade dos membros da classe base nas classes derivadas.

A classe base é a classe principal na hierarquia e serve como um modelo para as classes derivadas. Ela contém atributos e métodos que são compartilhados por todas as classes que herdam dela. As classes derivadas, por sua vez, herdam esses membros e podem adicionar novos membros ou modificar o comportamento existente.

Existem diferentes tipos de herança em C++, incluindo herança pública, privada e protegida. Na herança pública, os membros públicos da classe base permanecem públicos nas classes derivadas. Na herança privada, os membros públicos e protegidos da classe base tornam-se privados nas classes derivadas. Já na herança protegida, os membros públicos e protegidos da classe base tornam-se protegidos nas classes derivadas.

Além disso, C++ suporta herança múltipla, onde uma classe pode herdar de mais de uma classe base. Isso proporciona uma flexibilidade adicional na modelagem de relacionamentos complexos entre classes.

Estude o código abaixo, pois ele mostra a construção ideal uma classe básica(pai) e uma classe derivada(filho). Ou seja a classe derivada herda caracteristicas da classe básica. Ou ponto importante é a forma como o construtor pode inicializar os membros de dados da classe(vasculhe o código para aprender a olhar para um programa feito por outra pessoa). O contrutor pode ser inicializado tanto na forma apresentada no construtor da classe básica quanto na forma da classe derivada. 

            
                 

                 
#include <iostream>
using namespace std;





//===========================================
//--classe básica--

        class Aeronaves
        {
                
        public:
            Aeronaves(int=0, string="", string="", string="");//cnt
                
            void setNumeroPassageiros(int);
            int  getNumeroPassageiros();
                
            void setNomeAeronave(string);
            string  getNomeAeronave();
                    
            void   setTipoAeronave(string);
            string getTipoAeronave();
                  
            void   setHibrido(string);
            string getHibrido();
                    
        private:
            int _Npassageiros;
            string _TipoAeronave;
            string _Hibrido;
            string _NomeAeronave;
                  
        };//Fim classe Aeronaves






        //===========================================
        //--classe Derivada--
            
        class fokker:public Aeronaves
        {
        public:
            fokker(float=0.0, int=0, string="", string="", string="");//cnt
                   
            void  setvelocidade(float);
            float getvelocidade();
                
            void display();//mostra as informações                       
                    
        private:
            float _velocidade;
                    
        };//Fim da classe fokker

        

            
        int main()
        {
             cout << "Hello world!" << endl;
                    //objeto
                    fokker M5;//crio um objeto M5, instanciado da classe fokker
                    M5.setNumeroPassageiros(151);//seto o numero de passageiros
                    M5.setHibrido("Nao");//seto o a informação de é hibrido ou não
                    M5.setTipoAeronave("Comercial");//seto a classificação dele
                    M5.setvelocidade(900.0);//seto a velocidade
                    M5.setNomeAeronave("Fokker100");//seto o nome
                    M5.display();//nesta função-membro uso para mostrar todas as informações
                  
                    cout<<"\n"<<endl;//crio um espaço
                     
                    //Crio outro objeto chamado spin instanciado da classe fokker
                    //velocidade menor que 850.0 então é setado para 850.0                        
                    fokker spin(10.1, 1, "Combate", "Nao", "Fokker_spin");
                    spin.display();//Mostro as info.
                    
                     return 0;
                }

                       


 

                   
                //===========================================
                //--Desenvolvimento das funçoes da classe básica--

                Aeronaves::Aeronaves(int Npassageiro, string TipoAeronave, string Hibrido, string nome)
                {
                    _Npassageiros=Npassageiro;
                    _TipoAeronave=TipoAeronave;
                    _Hibrido=Hibrido;
                    _NomeAeronave=nome;
                }
                void Aeronaves::setNumeroPassageiros(int passgeiros)//seto o numero de passageiros restringindo o minímo
                {
                    
                //uso o perador ternário para evitar de usar o if-else
                    _Npassageiros=(passgeiros <0 && 340 ? 0: passgeiros);
                }
                int Aeronaves::getNumeroPassageiros()//retorno o atributo.
                {
                    return _Npassageiros;
                }
                void Aeronaves::setNomeAeronave(string nome)//seto o nome a aeronave
                {
                    _NomeAeronave=nome;
                }
                string Aeronaves::getNomeAeronave()//devolvo o atributo inicializado
                {
                    return _NomeAeronave;
                }
                        
                void  Aeronaves::setTipoAeronave(string TipoAero)//seto o tipo de aeronave
                {
                    _TipoAeronave=TipoAero;
                }
                string Aeronaves::getTipoAeronave()//devolvo o atributo inicializado
                {
                    return _TipoAeronave;
                }
                void  Aeronaves::setHibrido(string Hib)//seto se é hibrido
                {
                    _Hibrido=Hib;
                }
                string Aeronaves::getHibrido()//devolvo o atributo inicializado
                {
                    return _Hibrido;
                }

                _velocidade(velocidade<850.0? 850.0: velocidade),
                Aeronaves(passageiros, tipoAeronaves, hibrido, nome)//chamo construtor da classe básica para inicializar

                {
                    return _velocidade;
                }
